Paste and run this query inside your Supabase SQL Editor to create tables:

create table applications (
  id text primary key,
  company text not null,
  role text,
  stage text,
  priority integer,
  source text,
  location text,
  salary text,
  link text,
  resume text,
  notes text,
  contacts jsonb default '[]'::jsonb,
  timeline jsonb default '[]'::jsonb,
  lastUpdated text
);

create table activities (
  id text primary key,
  text text,
  date text
);

create table resumes (
  id text primary key,
  name text,
  target text,
  url text
);

alter table applications disable row level security;
alter table activities disable row level security;
alter table resumes disable row level security;
